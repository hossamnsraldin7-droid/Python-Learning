Student Management System 



# نظام إدارة درجات الطلاب - مشروع شامل يجمع كل مفاهيم المذكرة
students_database = {}

def add_student(name, grade):
    students_database[name] = grade
    print(f"تم تسجيل الطالب [{name}] بنجاح بدرجة {grade}.")

def generate_report_and_stats():
    if not students_database:
        print("قاعدة البيانات فارغة تماماً، لا يوجد طلاب مسجلون.")
        return
    
    grades = list(students_database.values())
    avg_grade = sum(grades) / len(grades)
    max_grade = max(grades)
    min_grade = min(grades)
    
    print("\n" + "="*30)
    print(f" المتوسط العام للدرجات: {avg_grade:.1f}")
    print(f" أعلى درجة في الفصل: {max_grade}")
    print(f" أدنى درجة في الفصل: {min_grade}")
    print("="*30)
    
    # حفظ التقرير تفصيلياً في ملف نصي خارجي بشكل دائم
    with open("students_final_report.txt", "w", encoding="utf-8") as f:
        f.write("=== تقرير نتائج الطلاب النهائي ===\n")
        for name, grade in students_database.items():
            f.write(f"الطالب: {name} --------> الدرجة: {grade}\n")
        f.write("\n" + "-"*30 + "\n")
        f.write(f"المعدل العام: {avg_grade:.1f} | أعلى درجة: {max_grade} | أدنى درجة: {min_grade}\n")
    print("\n تم تصدير وحفظ التقرير بنجاح في ملف: students_final_report.txt")

# --- تجربة تشغيل النظام العملي ---
add_student("محمد", 95)
add_student("سارة", 88)
add_student("علي", 72)

# استخراج التقارير الرياضية وحفظ الملف
generate_report_and_stats()
